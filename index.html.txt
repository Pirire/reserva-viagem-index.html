  <script>
    const stripe = Stripe("pk_test_51S5DWFRp3yrCaO94FKa6rZ2YBUVSQ6ykSzx1GnRW5zmFCarNb1Wce1GhpAM3uEpmKLhnsTubGqC4o0hr8UMVdsj000SwLnSI13");
    const backendURL = "https://reserva-5gm0.onrender.com"; 

    const btns = document.querySelectorAll('.btn-cat');
    let precoCategoria = 0, categoriaSelecionada = "", ultimoCusto = 0, ultimaDistancia = 0;
    let partidaPlace = null, destinoPlace = null;

    btns.forEach(btn => {
      btn.className = "btn-cat w-full py-2 rounded-2xl bg-green-300 hover:bg-green-400";
      btn.addEventListener('click', () => {
        btns.forEach(b => { b.classList.remove('bg-green-600'); b.classList.add('bg-green-300'); });
        btn.classList.remove('bg-green-300'); btn.classList.add('bg-green-600');
        precoCategoria = parseFloat(btn.getAttribute("data-preco"));
        categoriaSelecionada = btn.getAttribute("data-nome");
        validarFormulario(); calcularRotaDebounced();
      });
    });

    const camposObrig = [document.getElementById('partida'), document.getElementById('destino'), document.getElementById('datahora'), document.getElementById('nome')];
    camposObrig.forEach(c => c.addEventListener('input', () => { validarFormulario(); calcularRotaDebounced(); }));

    function validarFormulario() {
      const categoriaOK = categoriaSelecionada !== "";
      const preenchidos = camposObrig.every(c => c.value.trim() !== "");
      const btn = document.getElementById('OK');
      if (categoriaOK && preenchidos) {
        btn.disabled = false;
        btn.className = "mt-3 w-full py-2 rounded-2xl bg-green-500 hover:bg-green-600 text-white font-bold";
      } else {
        btn.disabled = true;
        btn.className = "mt-3 w-full py-2 rounded-2xl bg-gray-400 text-white font-bold";
      }
    }

    let map, directionsService, directionsRenderer;
    function initAutocomplete() {
      const partidaInput = document.getElementById('partida');
      const destinoInput = document.getElementById('destino');

      const partidaAC = new google.maps.places.Autocomplete(partidaInput, { fields: ["place_id", "formatted_address", "geometry"], componentRestrictions: { country: ["pt", "es", "fr"] } });
      const destinoAC = new google.maps.places.Autocomplete(destinoInput, { fields: ["place_id", "formatted_address", "geometry"], componentRestrictions: { country: ["pt", "es", "fr"] } });

      partidaAC.addListener('place_changed', () => { const p = partidaAC.getPlace(); if (p && p.place_id) { partidaPlace = p; partidaInput.value = p.formatted_address || partidaInput.value; calcularRotaDebounced(); } });
      destinoAC.addListener('place_changed', () => { const d = destinoAC.getPlace(); if (d && d.place_id) { destinoPlace = d; destinoInput.value = d.formatted_address || destinoInput.value; calcularRotaDebounced(); } });
      initMap();
    }

    function initMap() {
      map = new google.maps.Map(document.getElementById("map"), { zoom: 7, center: { lat: 38.7169, lng: -9.139 } });
      directionsService = new google.maps.DirectionsService();
      directionsRenderer = new google.maps.DirectionsRenderer({ preserveViewport: false });
      directionsRenderer.setMap(map);
    }

    let rotaTimer = null;
    function calcularRotaDebounced() { clearTimeout(rotaTimer); rotaTimer = setTimeout(calcularRota, 400); }
    function calcularRota() {
      const partidaTxt = document.getElementById("partida").value.trim();
      const destinoTxt = document.getElementById("destino").value.trim();
      if (!partidaTxt || !destinoTxt) return;

      const origin = (partidaPlace && partidaPlace.place_id) ? { placeId: partidaPlace.place_id } : partidaTxt;
      const destination = (destinoPlace && destinoPlace.place_id) ? { placeId: destinoPlace.place_id } : destinoTxt;

      directionsService.route({ origin, destination, travelMode: google.maps.TravelMode.DRIVING, region: "PT" }, (result, status) => {
        const info = document.getElementById("infoViagem");
        const box = document.getElementById("resultado");
        if (status === "OK") {
          directionsRenderer.setDirections(result);
          let metros = 0; result.routes[0].legs.forEach(l => metros += l.distance.value);
          ultimaDistancia = metros / 1000;
          if (precoCategoria > 0) { ultimoCusto = (ultimaDistancia * precoCategoria).toFixed(2); info.textContent = `Dist√¢ncia: ${ultimaDistancia.toFixed(2)} km | Valor Total: ${ultimoCusto} ‚Ç¨`; }
          else { ultimoCusto = 0; info.textContent = `Dist√¢ncia: ${ultimaDistancia.toFixed(2)} km`; }
          box.classList.remove("hidden");
        } else { info.textContent = `N√£o foi poss√≠vel calcular a rota (${status}).`; box.classList.remove("hidden"); }
      });
    }

    window.onload = initAutocomplete;

    document.getElementById('btnLocalizacao').addEventListener('click', () => {
      if (!navigator.geolocation) return alert('Geolocaliza√ß√£o n√£o suportada no seu navegador.');
      navigator.geolocation.getCurrentPosition(pos => {
        const geocoder = new google.maps.Geocoder();
        const latlng = { lat: pos.coords.latitude, lng: pos.coords.longitude };
        geocoder.geocode({ location: latlng }, (results, status) => {
          if (status === 'OK' && results[0]) { document.getElementById('partida').value = results[0].formatted_address; partidaPlace = null; validarFormulario(); calcularRotaDebounced(); }
          else alert('N√£o foi poss√≠vel obter o endere√ßo da sua localiza√ß√£o.');
        });
      });
    });

    document.getElementById("OK").addEventListener("click", () => {
      document.getElementById("conteudoPrincipal").classList.add("hidden");
      document.getElementById("resNome").textContent = document.getElementById("nome").value;
      document.getElementById("resCategoria").textContent = categoriaSelecionada;
      const partida = document.getElementById("partida").value.trim();
      const destino = document.getElementById("destino").value.trim();
      document.getElementById("resPartida").textContent = partida;
      document.getElementById("resPartida").href = `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(partida)}`;
      document.getElementById("resDestino").textContent = destino;
      document.getElementById("resDestino").href = `https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(destino)}`;
      document.getElementById("resData").textContent = document.getElementById("datahora").value;
      document.getElementById("resValor").textContent = ultimoCusto > 0 ? `${ultimoCusto} ‚Ç¨` : "‚Äî";
      document.getElementById("popupResumo").classList.remove("hidden");
    });

    function resetTudo() {
      document.getElementById("popupResumo").classList.add("hidden");
      document.getElementById("conteudoPrincipal").classList.remove("hidden");
      ["nome", "partida", "destino", "datahora"].forEach(id => document.getElementById(id).value = "");
      document.getElementById("infoViagem").textContent = ""; document.getElementById("resultado").classList.add("hidden");
      btns.forEach(b => { b.classList.remove('bg-green-600'); b.classList.add('bg-green-300'); });
      precoCategoria = 0; categoriaSelecionada = ""; ultimoCusto = 0; ultimaDistancia = 0; partidaPlace = null; destinoPlace = null;
      validarFormulario();
      map.setCenter({ lat: 38.7169, lng: -9.139 }); map.setZoom(7); directionsRenderer.set('directions', null);
    }

    document.getElementById("confirmarWhatsApp").addEventListener("click", () => {
      const mensagem = `üöñ Reserva de Viagem\n\nüë§ Nome: ${document.getElementById("resNome").textContent}\nüöó Categoria: ${document.getElementById("resCategoria").textContent}\n\nüìç Partida: ${document.getElementById("resPartida").textContent}\nüèÅ Destino: ${document.getElementById("resDestino").textContent}\nüìÖ Data: ${document.getElementById("resData").textContent}\nüí∂ Valor: ${document.getElementById("resValor").textContent}${ultimaDistancia ? `\nüìè Dist√¢ncia: ${ultimaDistancia.toFixed(2)} km` : ""}`;
      window.open(`https://wa.me/351938070495?text=${encodeURIComponent(mensagem)}`, "_blank");
      resetTudo();
    });

    document.getElementById("confirmarEmail").addEventListener("click", () => {
      const body = `Reserva de Viagem\nNome: ${document.getElementById("resNome").textContent}\nCategoria: ${document.getElementById("resCategoria").textContent}\nPartida: ${document.getElementById("resPartida").textContent}\nDestino: ${document.getElementById("resDestino").textContent}\nData: ${document.getElementById("resData").textContent}\nValor: ${document.getElementById("resValor").textContent}`;
      window.location.href = `mailto:?subject=Reserva&body=${encodeURIComponent(body)}`;
      resetTudo();
    });

    document.getElementById("confirmarPagamento").addEventListener("click", async () => {
      try {
        const valor = document.getElementById("resValor").textContent.replace(" ‚Ç¨", "");
        const nome = document.getElementById("resNome").textContent;
        const partida = document.getElementById("resPartida").textContent;
        const destino = document.getElementById("resDestino").textContent;
        const data = document.getElementById("resData").textContent;

        const res = await fetch(backendURL, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ valor, nome, partida, destino, data })
        });
        const dataResp = await res.json();
        window.location.href = dataResp.url;
      } catch (err) {
        alert("Erro ao processar pagamento: " + err.message);
      }
    });
  </script>
